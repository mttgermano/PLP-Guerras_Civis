# Create User
curl -X POST -H "Content-Type: application/json" -d '{"uName":"john_doe", "uPassword":"password123"}' http://localhost:3000/login/create_user/

# Login User
curl -X POST -H "Content-Type: application/json" -d '{"uName":"john_doe", "uPassword":"password123"}' http://localhost:3000/login/login_user/

# Login Room
